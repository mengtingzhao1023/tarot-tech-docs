2.1.2 自动生成编号算法

/**
 * 生成牌组编号
 * 规则：按购买日期排序，最早为第1副
 */
function generateSequenceNumber(newDeck, existingDecks) {
  // 1. 获取新牌组的购买日期
  const newPurchaseDate = new Date(newDeck.archive.purchaseDate);
  
  // 2. 将所有牌组（包括新的）按购买日期排序
  const allDecks = [...existingDecks, newDeck];
  allDecks.sort((a, b) => {
    return new Date(a.archive.purchaseDate) - new Date(b.archive.purchaseDate);
  });
  
  // 3. 找到新牌组在排序后的位置
  const index = allDecks.findIndex(d => d.id === newDeck.id);
  
  // 4. 生成编号（从1开始）
  return index + 1;
}

2.1.3 牌组CRUD操作

class DeckService {
  /**
   * 创建牌组
   */
  static async createDeck(deckData) {
    // 1. 生成唯一ID
    const deckId = `deck_${Date.now()}`;
    
    // 2. 处理图片（压缩、转base64）
    const processedImages = await ImageService.processImages(deckData.images);
    
    // 3. 生成编号
    const existingDecks = await this.getAllDecks();
    const sequenceNumber = generateSequenceNumber(
      { ...deckData, id: deckId },
      existingDecks
    );
    
    // 4. 构建完整数据
    const deck = {
      id: deckId,
      sequenceNumber,
      ...deckData,
      images: processedImages,
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString()
    };
    
    // 5. 保存到 IndexedDB
    await IndexedDBService.saveDeck(deck);
    
    // 6. 异步同步到 Supabase
    SyncService.syncToSupabase('decks', deck).catch(console.error);
    
    return deck;
  }
  
  /**
   * 删除牌组（软删除）
   */
  static async deleteDeck(deckId) {
    // 软删除，保留数据关联
    await this.updateDeck(deckId, {
      status: { isArchived: true }
    });
  }
}

2.2 牌灵绑定功能

class SpiritSelector {
  /**
   * 加载22张大阿卡纳牌
   */
  static async loadSpirits() {
    // 从牌灵总览数据获取
    const spirits = await fetch('/data/tarotCards.json')
      .then(res => res.json())
      .then(data => data.majorArcana);
    
    return spirits.map(card => ({
      id: card.id,
      name: card.name,
      englishName: card.englishName,
      keywords: card.keywords,
      element: card.element,
      number: card.number,
      image: card.image,
      description: card.description
    }));
  }
  
  /**
   * 随机抽取牌灵
   */
  static drawRandomSpirit(spirits) {
    const randomIndex = Math.floor(Math.random() * spirits.length);
    return spirits[randomIndex];
  }
  
  /**
   * 绑定牌灵到牌组
   */
  static async bindSpirit(deckId, spiritId, bondType = 'chosen') {
    const spirit = await this.getSpiritById(spiritId);
    
    return await DeckService.updateDeck(deckId, {
      spirit: {
        cardId: spirit.id,
        cardName: spirit.name,
        bondedAt: new Date().toISOString(),
        bondType
      }
    });
  }
}

2.3 图片处理功能

class ImageService {
  /**
   * 压缩图片
   */
  static async compressImage(file, maxWidth = 800, quality = 0.8) {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onload = (e) => {
        const img = new Image();
        img.onload = () => {
          const canvas = document.createElement('canvas');
          const ctx = canvas.getContext('2d');
          
          // 计算压缩后的尺寸
          let width = img.width;
          let height = img.height;
          if (width > maxWidth) {
            height = (maxWidth / width) * height;
            width = maxWidth;
          }
          
          canvas.width = width;
          canvas.height = height;
          ctx.drawImage(img, 0, 0, width, height);
          
          // 转为 base64
          const compressed = canvas.toDataURL('image/jpeg', quality);
          resolve(compressed);
        };
        img.src = e.target.result;
      };
      reader.onerror = reject;
      reader.readAsDataURL(file);
    });
  }
  
  /**
   * 估算存储大小
   */
  static estimateSize(base64String) {
    const bytes = (base64String.length * 3) / 4;
    return bytes / 1024 / 1024; // MB
  }
}

3.1 使用记录

class UsageService {
  /**
   * 记录牌组使用
   */
  static async recordUsage(deckId, usageData) {
    const usage = {
      id: `usage_${Date.now()}`,
      deckId,
      timestamp: new Date().toISOString(),
      ...usageData
    };
    
    // 保存使用记录
    await IndexedDBService.saveUsage(usage);
    
    // 更新牌组统计
    await this.updateDeckStats(deckId, usage);
    
    return usage;
  }
  
  /**
   * 计算30天趋势
   */
  static async calculateTrend(deckId) {
    const usages = await IndexedDBService.getUsagesByDeck(deckId, 30);
    const trend = new Array(30).fill(0);
    
    const now = new Date();
    usages.forEach(usage => {
      const daysAgo = Math.floor((now - new Date(usage.timestamp)) / (1000 * 60 * 60 * 24));
      if (daysAgo < 30) {
        trend[29 - daysAgo]++;
      }
    });
    
    return trend;
  }
}

四、统计与提醒功能

4.1 低频使用提醒

class ReminderService {
  /**
   * 检查需要提醒的牌组
   */
  static async checkReminders() {
    const decks = await DeckService.getAllDecks();
    const reminders = [];
    
    const now = new Date();
    
    decks.forEach(deck => {
      if (!deck.stats.reminder.enabled) return;
      if (deck.status.isArchived) return;
      
      const lastUsed = new Date(deck.stats.lastUsedAt);
      const daysSinceLastUse = Math.floor((now - lastUsed) / (1000 * 60 * 60 * 24));
      const threshold = deck.stats.reminder.thresholdDays;
      
      if (daysSinceLastUse >= threshold) {
        reminders.push({
          deckId: deck.id,
          deckName: deck.name,
          sequenceNumber: deck.sequenceNumber,
          spiritName: deck.spirit.cardName,
          daysSinceLastUse,
          lastUsedAt: deck.stats.lastUsedAt
        });
      }
    });
    
    return reminders;
  }
}

class SpiritRelationService {
  /**
   * 获取牌组的完整牌灵信息
   */
  static async getSpiritDetail(deckId) {
    const deck = await DeckService.getDeckById(deckId);
    const spiritCardId = deck.spirit.cardId;
    
    // 从牌灵总览获取完整数据
    const tarotCards = await fetch('/data/tarotCards.json')
      .then(res => res.json());
    
    const spiritCard = tarotCards.majorArcana.find(c => c.id === spiritCardId);
    
    return {
      ...deck.spirit,
      ...spiritCard,
      deckId
    };
  }
  
  /**
   * 获取拥有某牌灵的所有牌组
   */
  static async getDecksBySpirit(spiritCardId) {
    const decks = await DeckService.getAllDecks();
    return decks.filter(d => d.spirit.cardId === spiritCardId);
  }
}

六、数据同步设计

class SyncService {
  /**
   * 同步到 Supabase
   */
  static async syncToSupabase(table, data) {
    try {
      const { error } = await supabase
        .from(table)
        .upsert(data, { onConflict: 'id' });
      
      if (error) throw error;
      
      console.log(`Synced ${table}:`, data.id);
    } catch (err) {
      console.error('Sync failed:', err);
      // 标记待同步
      await this.markPendingSync(table, data);
    }
  }
  
  /**
   * 合并本地和云端数据（解决冲突）
   */
  static mergeDecks(local, cloud) {
    const merged = new Map();
    
    // 添加本地数据
    local.forEach(deck => merged.set(deck.id, deck));
    
    // 合并云端数据（以更新时间较新的为准）
    cloud.forEach(cloudDeck => {
      const localDeck = merged.get(cloudDeck.id);
      if (!localDeck) {
        merged.set(cloudDeck.id, cloudDeck);
      } else {
        const localTime = new Date(localDeck.updatedAt);
        const cloudTime = new Date(cloudDeck.updatedAt);
        if (cloudTime > localTime) {
          merged.set(cloudDeck.id, cloudDeck);
        }
      }
    });
    
    return Array.from(merged.values());
  }
}

// 图片懒加载
class LazyImageLoader {
  static init() {
    const imageObserver = new IntersectionObserver((entries, observer) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const img = entry.target;
          img.src = img.dataset.src;
          img.classList.remove('lazy');
          observer.unobserve(img);
        }
      });
    });
    
    document.querySelectorAll('img.lazy').forEach(img => {
      imageObserver.observe(img);
    });
  }
}

八、错误处理

class ErrorHandler {
  static handle(error, context) {
    console.error(`Error in ${context}:`, error);
    
    switch (error.type) {
      case 'storage_full':
        this.showMessage('存储空间不足，请删除一些图片');
        break;
      case 'network_error':
        this.showMessage('网络连接失败，已保存到本地');
        break;
      case 'sync_conflict':
        this.showMessage('数据冲突，已自动合并');
        break;
      default:
        this.showMessage('操作失败，请重试');
    }
  }
}
