3.1 表结构

-- ==================== 牌组表 ====================
CREATE TABLE decks (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    
    -- 编号（自动生成）
    sequence_number INTEGER NOT NULL,
    
    -- 基本信息
    name TEXT NOT NULL,
    preset_id TEXT,
    
    -- 牌灵信息
    spirit_card_id TEXT NOT NULL,
    spirit_card_name TEXT NOT NULL,
    spirit_bonded_at TIMESTAMPTZ DEFAULT NOW(),
    spirit_bond_type TEXT CHECK (spirit_bond_type IN ('chosen', 'drawn')),
    
    -- 图片（存储URL或base64标识）
    image_cover TEXT,
    image_back TEXT,
    image_samples JSONB DEFAULT '[]',
    
    -- 档案信息
    archive_purchase_date DATE,
    archive_purchase_price INTEGER,
    archive_condition TEXT CHECK (archive_condition IN ('brand_new', 'good', 'fair', 'worn')),
    archive_card_count INTEGER DEFAULT 78,
    archive_size TEXT,
    archive_material TEXT,
    archive_notes TEXT,
    
    -- 性格档案
    personality_description TEXT,
    personality_traits JSONB DEFAULT '[]',
    personality_user_tags JSONB DEFAULT '[]',
    
    -- 状态
    status_is_pinned BOOLEAN DEFAULT false,
    status_is_active BOOLEAN DEFAULT true,
    status_is_archived BOOLEAN DEFAULT false,
    
    -- 元数据
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_decks_user_id ON decks(user_id);
CREATE INDEX idx_decks_sequence ON decks(user_id, sequence_number);
CREATE INDEX idx_decks_spirit ON decks(spirit_card_id);
CREATE INDEX idx_decks_pinned ON decks(user_id, status_is_pinned) WHERE status_is_pinned = true;

-- ==================== 使用记录表 ====================
CREATE TABLE deck_usages (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    deck_id UUID REFERENCES decks(id) ON DELETE CASCADE,
    user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    
    -- 使用信息
    timestamp TIMESTAMPTZ DEFAULT NOW(),
    spread_id TEXT,
    spread_name TEXT,
    question TEXT,
    
    -- 抽牌结果
    cards JSONB NOT NULL, -- [{name, position, orientation}, ...]
    
    -- 关联
    draw_record_id UUID,
    client_id UUID,
    
    -- 环境信息
    mood TEXT,
    environment TEXT,
    notes TEXT,
    
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_usages_deck ON deck_usages(deck_id);
CREATE INDEX idx_usages_timestamp ON deck_usages(deck_id, timestamp DESC);
CREATE INDEX idx_usages_client ON deck_usages(client_id) WHERE client_id IS NOT NULL;

-- ==================== 对话记录表 ====================
CREATE TABLE deck_conversations (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    deck_id UUID REFERENCES decks(id) ON DELETE CASCADE,
    user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    
    -- 类型：dialogue(与牌对话) | case(抽牌案例)
    type TEXT CHECK (type IN ('dialogue', 'case')),
    
    -- 基本信息
    date TIMESTAMPTZ DEFAULT NOW(),
    question TEXT,
    answer TEXT,
    
    -- 抽到的牌
    cards JSONB DEFAULT '[]',
    
    -- 案例特有
    client_id UUID,
    client_name TEXT,
    spread_id TEXT,
    spread_name TEXT,
    draw_record_id UUID,
    
    -- 心情
    mood TEXT,
    
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_conversations_deck ON deck_conversations(deck_id);
CREATE INDEX idx_conversations_type ON deck_conversations(deck_id, type);
CREATE INDEX idx_conversations_date ON deck_conversations(deck_id, date DESC);

-- ==================== 使用统计表（物化视图）====================
CREATE MATERIALIZED VIEW deck_stats AS
SELECT 
    deck_id,
    COUNT(*) as total_uses,
    MIN(timestamp) as first_used_at,
    MAX(timestamp) as last_used_at,
    COUNT(*) FILTER (WHERE timestamp >= NOW() - INTERVAL '7 days') as weekly_uses,
    COUNT(*) FILTER (WHERE timestamp >= NOW() - INTERVAL '30 days') as monthly_uses,
    mode() WITHIN GROUP (ORDER BY spread_id) as favorite_spread
FROM deck_usages
GROUP BY deck_id;

-- 刷新统计视图的函数
CREATE OR REPLACE FUNCTION refresh_deck_stats()
RETURNS void AS $$
BEGIN
    REFRESH MATERIALIZED VIEW deck_stats;
END;
$$ LANGUAGE plpgsql;

// ==================== 牌组 API ====================

/**
 * 获取用户的所有牌组
 * GET /rest/v1/decks?select=*&user_id=eq.{userId}&order=sequence_number.asc
 */
const getDecks = async () => {
  const { data, error } = await supabase
    .from('decks')
    .select('*')
    .order('sequence_number', { ascending: true });
  
  return { data, error };
};

/**
 * 获取单个牌组
 * GET /rest/v1/decks?select=*&id=eq.{deckId}
 */
const getDeckById = async (deckId) => {
  const { data, error } = await supabase
    .from('decks')
    .select(`
      *,
      deck_usages (count),
      deck_conversations (*)
    `)
    .eq('id', deckId)
    .single();
  
  return { data, error };
};

/**
 * 创建牌组
 * POST /rest/v1/decks
 */
const createDeck = async (deckData) => {
  // 1. 获取当前最大编号
  const { data: maxSeq } = await supabase
    .from('decks')
    .select('sequence_number')
    .order('sequence_number', { ascending: false })
    .limit(1)
    .single();
  
  const sequenceNumber = (maxSeq?.sequence_number || 0) + 1;
  
  // 2. 创建牌组
  const { data, error } = await supabase
    .from('decks')
    .insert([{
      ...deckData,
      sequence_number: sequenceNumber,
      user_id: supabase.auth.user().id
    }])
    .select()
    .single();
  
  return { data, error };
};

/**
 * 置顶/取消置顶
 * PATCH /rest/v1/decks?id=eq.{deckId}
 */
const togglePinDeck = async (deckId, isPinned) => {
  return await updateDeck(deckId, {
    status_is_pinned: isPinned
  });
};

4.2 Realtime 订阅

// ==================== 实时订阅 ====================

class RealtimeService {
  constructor() {
    this.subscriptions = new Map();
  }
  
  /**
   * 订阅牌组变化
   */
  subscribeToDecks(callback) {
    const subscription = supabase
      .channel('decks_changes')
      .on(
        'postgres_changes',
        {
          event: '*',
          schema: 'public',
          table: 'decks',
          filter: `user_id=eq.${supabase.auth.user().id}`
        },
        (payload) => {
          callback(payload);
        }
      )
      .subscribe();
    
    this.subscriptions.set('decks', subscription);
    return subscription;
  }
  
  /**
   * 订阅使用记录变化
   */
  subscribeToUsages(deckId, callback) {
    const subscription = supabase
      .channel(`usages_${deckId}`)
      .on(
        'postgres_changes',
        {
          event: 'INSERT',
          schema: 'public',
          table: 'deck_usages',
          filter: `deck_id=eq.${deckId}`
        },
        (payload) => {
          callback(payload.new);
        }
      )
      .subscribe();
    
    this.subscriptions.set(`usages_${deckId}`, subscription);
    return subscription;
  }
}

5.1 核心服务实现

// services/deck/DeckService.js

import { supabase } from '../../config/supabase.js';
import { DeckStorageService } from './DeckStorageService.js';
import { DeckSyncService } from './DeckSyncService.js';

export class DeckService {
  /**
   * 获取所有牌组（离线优先）
   */
  static async getAllDecks() {
    // 1. 先从本地获取
    let decks = await DeckStorageService.getAllFromLocal();
    
    // 2. 如果本地没有，从云端获取
    if (!decks || decks.length === 0) {
      const { data, error } = await supabase
        .from('decks')
        .select('*')
        .eq('status_is_archived', false)
        .order('sequence_number', { ascending: true });
      
      if (error) throw error;
      
      decks = data;
      
      // 保存到本地
      await DeckStorageService.saveAllToLocal(decks);
    }
    
    return decks;
  }
  
  /**
   * 创建牌组
   */
  static async createDeck(deckData) {
    // 1. 处理图片
    const processedImages = await this.processImages(deckData.images);
    
    // 2. 生成编号
    const decks = await this.getAllDecks();
    const sequenceNumber = this.generateSequenceNumber(
      deckData.archive.purchaseDate,
      decks
    );
    
    // 3. 构建数据
    const deck = {
      ...deckData,
      sequence_number: sequenceNumber,
      images: processedImages,
      user_id: supabase.auth.user().id
    };
    
    // 4. 保存到云端
    const { data, error } = await supabase
      .from('decks')
      .insert([deck])
      .select()
      .single();
    
    if (error) throw error;
    
    // 5. 保存到本地
    await DeckStorageService.saveToLocal(data);
    
    return data;
  }
  
  /**
   * 压缩图片
   */
  static async compressImage(file, maxWidth) {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onload = (e) => {
        const img = new Image();
        img.onload = () => {
          const canvas = document.createElement('canvas');
          const ctx = canvas.getContext('2d');
          
          let width = img.width;
          let height = img.height;
          
          if (width > maxWidth) {
            height = (maxWidth / width) * height;
            width = maxWidth;
          }
          
          canvas.width = width;
          canvas.height = height;
          ctx.drawImage(img, 0, 0, width, height);
          
          resolve(canvas.toDataURL('image/jpeg', 0.8));
        };
        img.src = e.target.result;
      };
      reader.onerror = reject;
      reader.readAsDataURL(file);
    });
  }
}

      img.src = img.dataset.src;
      img.classList.remove('lazy');
      imageObserver.unobserve(img);
    }
  });
});

// Service Worker 缓存
const CACHE_NAME = 'tarot-app-v1';

self.addEventListener('fetch', (event) => {
  event.respondWith(
    caches.match(event.request).then((response) => {
      // 缓存优先
      if (response) {
        return response;
      }
      
      return fetch(event.request).then((response) => {
        // 缓存新请求
        if (response.status === 200) {
          const responseClone = response.clone();
          caches.open(CACHE_NAME).then((cache) => {
            cache.put(event.request, responseClone);
          });
        }
        return response;
      });
    })
  );
});
